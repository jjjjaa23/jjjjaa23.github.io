<?php
header('Content-Type: application/json; charset=utf-8');

if ($_SERVER['REQUEST_METHOD'] !== 'POST') {
    echo json_encode(['status' => 'error', 'message' => 'Invalid Request Method']);
    exit;
}

// 1. 接收并清理表单输入，防止恶意注入
$name = filter_var(trim($_POST['name'] ?? ''), FILTER_SANITIZE_STRING);
$email = filter_var(trim($_POST['email'] ?? ''), FILTER_VALIDATE_EMAIL);
$company = filter_var(trim($_POST['company'] ?? ''), FILTER_SANITIZE_STRING);
$message = filter_var(trim($_POST['message'] ?? ''), FILTER_SANITIZE_STRING);

// 2. 基础验证
if (!$name || !$email || !$message) {
    echo json_encode(['status' => 'error', 'message' => 'Please fill in all required fields.']);
    exit;
}

// 3. 配置收件箱（改为您的实际接收邮箱）
$to = "sales@xiorve.com"; 
$subject = "🔥 XIORVE Web Inquiry from: $name ($company)";

// 4. 构建精美的邮件内容 HTML 格式
$email_content = "
<html>
<head>
    <title>New Lead from XIORVE Website</title>
</head>
<body style='font-family: Arial, sans-serif; line-height: 1.6; color: #333;'>
    <div style='max-width: 600px; margin: 0 auto; border: 1px solid #e2e8f0; border-radius: 8px; overflow: hidden;'>
        <div style='background-color: #0052cc; color: white; padding: 20px; text-align: center;'>
            <h2>New Inquiry Received</h2>
        </div>
        <div style='padding: 20px;'>
            <p><strong>Name:</strong> {$name}</p >
            <p><strong>Email:</strong> <a href=' '>{$email}</a ></p >
            <p><strong>Company:</strong> " . ($company ? $company : "Not Provided") . "</p >
            <hr style='border: 0; border-top: 1px solid #eee; margin: 20px 0;'>
            <p><strong>Inquiry Details / Message:</strong></p >
            <p style='background: #f8fafc; padding: 15px; border-left: 4px solid #0052cc; font-style: italic;'>
                " . nl2br($message) . "
            </p >
        </div>
        <div style='background-color: #f1f5f9; text-align: center; padding: 10px; font-size: 12px; color: #64748b;'>
            This inquiry was sent directly from the XIORVE website contact form.
        </div>
    </div>
</body>
</html>
";

// 5. 设置邮件头，确保支持 HTML 格式和 UTF-8
$headers = "MIME-Version: 1.0" . "\r\n";
$headers .= "Content-type:text/html;charset=UTF-8" . "\r\n";
$headers .= "From: XIORVE Web Robot <noreply@xiorve.com>" . "\r\n";
$headers .= "Reply-To: $email" . "\r\n";

// 6. 发送邮件并返回结果给前端 JS
if (mail($to, $subject, $email_content, $headers)) {
    echo json_encode(['status' => 'success', 'message' => 'Mail sent successfully']);
} else {
    echo json_encode(['status' => 'error', 'message' => 'Server failed to send email. Please try again.']);
}
